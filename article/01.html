web 调起摄像头 并截图保存

garMing 7/20/2022 article

由于是在 弹框中进行摄像 ，所以初始化的代码放在 updated中, 需要注意的是，在这个钩子中，当模版中的依赖被更新时，都会被触发 ，所以就需要加限制🚫


<script>
export default {
    data: {
      // 新增采集功能字段
      CaiJIan: true, // 是否裁剪
      shutter: null,
      canvas: null,
      context: null,
      video: null,
      mediaStreamTrack: null,
    },
    updated() {
        // 页面的所有依赖更新都会执行到这里，所以这里要异常小心
        if(this.mediaStreamTrack) return; // 当弹框关闭的时候，不去执行下面的逻辑
        this.video = this.$refs.videoRef
        if(this.video) {
        if (navigator.mediaDevices.getUserMedia || navigator.getUserMedia || navigator.webkitGetUserMedia || navigator.mozGetUserMedia) {
            //调用用户媒体设备, 访问摄像头
            this.getUserMedia({ video: this.video }, this.success, this.error);
        } else {
                alert('不支持访问用户媒体');
        }
        // 媒体的加载必须要引入一个媒体文件，否则不会生效
        this.shutter = new Audio('./shutter.mp3')
        //   初始化 canvas ref
        this.canvas = this.$refs.canvasRef 
        this.context = this.canvas.getContext('2d');
        }
    },
    methods: {
        getUserMedia(constraints, success, error) {
            if (navigator.mediaDevices.getUserMedia) {
                //最新的标准API
                navigator.mediaDevices.getUserMedia(constraints).then(success).catch(error);
            } else if (navigator.webkitGetUserMedia) {
                //webkit核心浏览器
                navigator.webkitGetUserMedia(constraints, success, error)
            } else if (navigator.mozGetUserMedia) {
                //firfox浏览器
                navigator.mozGetUserMedia(constraints, success, error);
            } else if (navigator.getUserMedia) {
                //旧版API
                navigator.getUserMedia(constraints, success, error);
            }
        },
        success(stream) {
            //兼容webkit核心浏览器
            // let CompatibleURL = window.URL || window.webkitURL;
            //将视频流设置为video元素的源
            //video.src = CompatibleURL.createObjectURL(stream);
            this.video.srcObject = stream;
            this.mediaStreamTrack = stream.getTracks()[0]
            this.video.play();
        },
        error(error) {
            console.log(`访问用户媒体设备失败${error.name}, ${error.message}`);
        },
            //彩图，拍照
    take_snapshot() {
      this.context.drawImage(this.video, 0, 0,  $(this.video).attr("width"), $(this.video).attr("height"));
      var img = new Image();
      img.src = this.canvas.toDataURL('image/jpeg', 1);

      if (this.CaiJIan) {
          img = this.caijianImg();
      }
        //   const file = this.base64ToFile(img.src, '图片采集')
        //   const URL = this.getImageUrl({
        //     status: 'ready',
        //     name: file.name,
        //     szie: file.size,
        //     percentage: 0,
        //     raw: file
        //   })
        //   this.imgList.push({
        //     img: URL,
        //     show: false
        //   })
        //   this.fileList.push({
        //     status: 'ready',
        //     name: file.name,
        //     szie: file.size,
        //     percentage: 0,
        //     raw: file
        //   })
    },
    //图片裁剪
    caijianImg() {
        var width = $(this.video).attr("width");
        var height = $(this.video).attr("height");

        // 整个采集图片向右偏移多少，也就是左边多少个像素不要了
        var sx = 100;
        // 整个采集图片向上偏移多少，也就是上边多少个像素不要了
        var sy = 100;
        //也就是右边多少个像素不要了
        var yby = 100;
        // 下边多少个像素不要了
        var xby = 100;


        var sWidth = width - sx;
        var sHeight = height - sy;

        var dWidth = width - sx;
        var dHeight = height - sy;

        var quality = 1;
        var dx = 0;
        var dy = 0;

        this.canvas.width = dWidth - yby;
        this.canvas.height = dHeight - xby;

        this.context.drawImage(this.video, sx, sy, sWidth, sHeight, dx, dy, dWidth, dHeight);

        var preview = new Image();
        preview.src = this.canvas.toDataURL('image/jpeg', quality);;
        return preview;
    },
    base64ToFile(urlData, fileName) {
      let arr = urlData.split(',');
      let mime = arr[0].match(/:(.*?);/)[1];
      let bytes = atob(arr[1]); // 解码base64
      let n = bytes.length
      let ia = new Uint8Array(n);
      while (n--) {
          ia[n] = bytes.charCodeAt(n);
      }
      return new File([ia], fileName, { type: mime });
    },
    getImageUrl(file) {
      let URL = null;
      if(window.createObjectURL !== undefined) {
        URL = window.createObjectURL(file.raw)
      } else if(window.URL !== undefined) {
        URL = window.URL.createObjectURL(file.raw)
      } else if(window.webkitURL !== undefined) {
        URL = window.webkitURL.createObjectURL(file.raw)
      }
      return URL
    }
  }
},
</script>
